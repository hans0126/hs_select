  • attrValue Top

    如果此參數未設定，則預設參數名稱為va,
    程式轉換過程中，所定義的第一個值將來自於option的value
    js. 
    	
			$("#select_example").hs_select({
				attrValue:['apple','orange','lemon'] //可代複數的自定義參數並於傳值階段傳遞出去
			});
    html. 
    			
			<select id="select_example" style="width:200px" >
				<option value="1" orange="2" lemon="3">item_1</option>
				<option value="4" orange="5" lemon="6">item_2</option>
				<option value="7" orange="8" lemon="9">item_3</option>
			</select>

  • arr_multi_select Top

    連動式下拉式選單，可以一對一或是一對多的Ajax連動
    js
    			//要使用連動時，不要使用ID呼叫jq，用class	
			//一對多 當元件A觸發時，元件B與元件C可同時連動
			$(".some_select").hs_select({
				 arr_multi_select:[{"current":"元件A","target_info":[{"target":"元件B","url":"目標網址一"},{"target":"元件C","url":"目標網址二"}]}]
			});	
			
			//一對一 當元件A觸發時，元件B連動，可定義多階層連動
			$(".some_select").hs_select({
				 arr_multi_select:[{"current":"元件A","target_info":[{"target":"元件B","url":"目標網址"}]},{"current":"元件B","target_info":[{"target":"元件C","url":"目標網址"}]
			});
    Json. 
    		<!-- 		
		{回傳的項目名稱:值,"value":{值1:"",值2:"",值3:""}
		須配合attrValue宣告值的名稱		
		 -->
		[{"title":"return item 1","value":{"apple":"1","orange":"2","lemon":"3"}}
		,{"title":"return item 2","value":{"apple":"4","orange":"5","lemon":"6"}}
		,{"title":"return item 3","value":{"apple":"7","orange":"8","lemon":"9"}}
		,{"title":"return item 4","value":{"apple":"10","orange":"11","lemon":"12"}}
		,{"title":"return item 5","value":{"apple":"13","orange":"14","lemon":"15"}}]

  • clickEvent Top

    項目選取事件
    js.
    						
			//clickEvent:function(所選取的值,當下select的物件)
			//若是attrValue無設定時，	預設va為一組值，若是有設定單組以上則會回傳object，使用方是如以下範例	
			
			$("#clickEventExample").hs_select({
					attrValue:['apple','orange','lemon'],
					li_padding:15,
					footer_corner:true,
					clickEvent:function(va,thisObj){						
						var out_text='所選的值<br/>';						
						for (var k in va) {							
							out_text+=k+":"+va[k]+"<br/>";							
						}		
						out_text+="ID="+thisObj.attr("id");
						$('#clickEventExampleText').html(out_text);											
					}
			});
    html.
    			
			<select id="select_example" style="width:200px" >
				<option value="1" orange="2" lemon="3">item_1</option>
				<option value="4" orange="5" lemon="6">item_2</option>
				<option value="7" orange="8" lemon="9">item_3</option>
			</select>

  • dataListHeight & drag_height_fix & drag_height Top

    卷軸
    js.
    			//dataListHeight 賦值時，若檔案列表超過設定高度，則會出現scroll bar				
			$("#scrollExample").hs_select({					
				li_padding:15,
				footer_corner:true,
				dataListHeight:150,
				drag_height:50	//控制drag 高度								
			});
				
			$("#scrollExample2").hs_select({					
				li_padding:15,
				footer_corner:true,
				dataListHeight:150,
				drag_height_fix:false //是否鎖定drag高度						
			});	
		
    			<!-- 以下是scroller的html碼，可至css定義樣式 -->
			<div class="track">
					<div class="drag ui-draggable"></div>
			</div>
    不論資料筆數多少drag會鎖住高度 drag高度會隨著資料筆數增減而有所變化  

  • data_reverse Top

  • theme Top

    在同一個頁面，使用不同的樣式
    js.
    			$("#themeExample").hs_select({					
				li_padding:15,
				footer_corner:true						
			});	
			
			$("#themeExample2").hs_select({					
				li_padding:15,
				theme:"example2",
				ul_width_tuning:-2,//因為兩邊多加了1px外邊框線條，因此整體寬度減少2px

			});
    html.

  • default_value_text & default_choice Top

    當未選擇時的預設值
    js.
    			/*當default_choice 關閉時，會帶入預設的-select- */
			$("#defaultChoice").hs_select({					
				li_padding:15,
				footer_corner:true,
				default_choice:false//不預先選擇某個項目而是用文自替代				
			});	
			/*可定義需要顯示的字*/
			$("#defaultChoice2").hs_select({					
				li_padding:15,
				footer_corner:true,
				default_choice:false,//不預先選擇某個項目而是用文自替代
				default_value_text:'-Please Choice-'				
			});	
			/*亦可定義需要顯示的字於select 物件內，如下面範例*/
			$("#defaultChoice3").hs_select({					
				li_padding:15,
				footer_corner:true,
				default_choice:false,//不預先選擇某個項目而是用文自替代
					
			});
